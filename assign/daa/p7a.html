int a[100],x[100];
int d;
void sum_of_subset (int weightsofar,int k,int remwt)
{
int i;
x[k] = 1;

{

if (x[i] == 1)

}
else if (weightsofar + a[k] + a[k+1] <= d)
sum_of_subset (weightsofar+a[k],k+1,remwt-a[k]);
